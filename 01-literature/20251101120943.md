# [Kap 1.5.8]([[20250818102829]]) Flettesortering
```java
private static void mergeSort(int[] a, int[] temp, int from, int to) {
    if (to - from <= 1) return;
    int mid = (from + to) / 2;
    mergeSort(a, temp, from, mid);
    mergeSort(a, temp, mid, to);
    merge(a, temp, from, mid, to);
}

public static void mergeSort(int[] a) {
    int[] temp = new int[a.length / 2];
    mergeSort(a, temp, 0, a.length);
}

private static void merge(int[] a, int[] temp, int from, int mid, int to) {
    int leftSize = mid - from;
    System.arraycopy(a, from, temp, 0, leftSize);

    int i = 0;
    int j = mid;
    int k = from;

    while (i < leftSize && j < to) {
        if (temp[i] <= a[j]) {
            a[k++] = temp[i++];
        } else {
            a[k++] = a[j++];
        }
    }

    while (i < leftSize) {
        a[k++] = temp[i++];
    }
}
```
