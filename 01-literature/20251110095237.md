# [4.1.3]([[20250818102829]]) Stack ved hjelp av en pekerkjede
Vi kan bruke en enkellistet liste som datastruktur under en stack.
```java
public class LinkedStack<T> implements Stack<T> {
	private static final class Node<T> {
    	private T value;
    	private Node<T> next;
        
    	private Node(T value, Node<T> next) {
    		this.value = value;
    		this.next = next;
    	}
    }
    
	private Node<T> head;
    private int n;

    public LinkedStack() {
		head = null;
        n = 0;
    }
    
    public void push(T value) {
        if (head == null) head = new Node<>(value, null)
    	else head = new Node<>(value, head);
        n++;
    }
    
    public T peek() {
        if (isEmpty()) throw new NoSuchElementException("The stack is empty");
        return head.value;
    }
    
    public T pop() {
        if (isEmpty()) throw new NoSuchElementException("The stack is empty");
        T val = head.value;
        Node<T> oldHead = head;
        head.value = null;
        head = head.next;
        oldHead.value = null;
        oldHead.next = null;
        n--;
        return val;
    }
    
    public int size() {
        return n;
    }
    
    public boolean isEmpty() {
        return n == 0;
    }

	public void clear() {
        while(head != null) {
            Node<T> next = head.next;
            head.value = null;
            head.next = null;
            head = next;
		}
        n = 0;
    }
}
```
